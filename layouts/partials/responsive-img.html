{{/* Responsive image helper. Processes an image from the asset pipeline into WebP variants at the given widths, emitting srcset + width/height (to avoid layout shift). Falls back to a plain if the source can't be processed. Params (dict): src - image path (e.g. "/images/comics/x.jpg" or "images/comics/x.jpg") alt - alt text class - optional class for the sizes - optional sizes attribute widths - optional slice of target widths (default: 400 800) loading - optional loading attr (default: "lazy") quality - optional WebP quality 1-100 (default: 82) */}} {{ $src := .src }} {{ $alt := .alt | default "" }} {{ $class := .class | default "" }} {{ $sizes := .sizes | default "" }} {{ $loading := .loading | default "lazy" }} {{ $widths := .widths | default (slice 400 800) }} {{ $quality := .quality | default 82 }} {{ $path := strings.TrimPrefix "/" $src }} {{ $res := resources.Get $path }} {{ if $res }} {{/* Cap every target at the original width (never upscale), then dedupe so small originals don't emit duplicate srcset entries. */}} {{ $targets := slice }} {{ range (sort $widths) }} {{ $targets = $targets | append (int (math.Min . $res.Width)) }} {{ end }} {{ $targets = uniq $targets }} {{ $srcset := slice }} {{ $main := "" }} {{ range $targets }} {{ $v := $res.Resize (printf "%dx webp q%d" . $quality) }} {{ $srcset = $srcset | append (printf "%s %dw" $v.RelPermalink (int $v.Width)) }} {{ $main = $v }} {{ end }} {{ else }} {{ end }}
